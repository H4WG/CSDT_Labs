int led_red = 0;        // the red LED is connected to Pin 0 of the Arduino.
int led_yellow = 1;     // the yellow LED is connected to Pin 1 of the Arduino.
int led_green = 2;      // the green LED is connected to Pin 2 of the Arduino.
int led_orange = 3;     // the orange LED is connected to Pin 3 of the Arduino.
int led_turqouise = 4;  // the turqouise LED is connected to Pin 4 of the Arduino.
int led_brown = 5;      // the brown LED is connected to Pin 5 of the Arduino.

void setup() {
  // set up all the LEDs as OUTPUT.
  pinMode(led_red, OUTPUT);
  pinMode(led_yellow, OUTPUT);
  pinMode(led_green, OUTPUT);
  pinMode(led_orange, OUTPUT);
  pinMode(led_turqouise, OUTPUT);
  pinMode(led_brown, OUTPUT);
}

void loop() {
  // turn the red LED on and the other LEDs off.
  digitalWrite(led_red, HIGH);
  digitalWrite(led_yellow, LOW);
  digitalWrite(led_green, LOW);
  digitalWrite(led_orange, LOW);
  digitalWrite(led_turqouise, LOW);
  digitalWrite(led_brown, LOW);
  delay(1000);    // wait 1 seconds.
  
  // turn the yellow LED on and the other LEDs off.
  digitalWrite(led_red, LOW);
  digitalWrite(led_yellow, HIGH);
  digitalWrite(led_green, LOW);
  digitalWrite(led_orange, LOW);
  digitalWrite(led_turqouise, LOW);
  digitalWrite(led_brown, LOW);
  delay(1000);   // wait 1 second.
  
  // turn the green LED on and the other LEDs off.
  digitalWrite(led_red, LOW);
  digitalWrite(led_yellow, LOW);
  digitalWrite(led_green, HIGH);
  digitalWrite(led_orange, LOW);
  digitalWrite(led_turqouise, LOW);
  digitalWrite(led_brown, LOW);
  delay(1000);  // wait 1 seconds.

  // turn the orange LED on and the other LEDs off.
  digitalWrite(led_red, LOW);
  digitalWrite(led_yellow, LOW);
  digitalWrite(led_green, LOW);
  digitalWrite(led_orange, HIGH);
  digitalWrite(led_turqouise, LOW);
  digitalWrite(led_brown, LOW);
  delay(1000);  // wait 1 seconds.
  
  // turn the turqouise LED on and the other LEDs off.
  digitalWrite(led_red, LOW);
  digitalWrite(led_yellow, LOW);
  digitalWrite(led_green, LOW);
  digitalWrite(led_orange, LOW);
  digitalWrite(led_turqouise, HIGH);
  digitalWrite(led_brown, LOW);
  delay(1000);  // wait 1 seconds.
  
  // turn the brown LED on and the other LEDs off.
  digitalWrite(led_red, LOW);  
  digitalWrite(led_yellow, LOW);
  digitalWrite(led_green, LOW);
  digitalWrite(led_orange, LOW);
  digitalWrite(led_turqouise, LOW);
  digitalWrite(led_brown, HIGH);
  delay(1000);  // wait 1 seconds.
}
